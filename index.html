/** * 🛰️ SYSTEM PROTOTYPE: QUANTUM TERMINAL V4.0 * -------------------------------------------------- * 遵照“绝不偷工减料”原则，全逻辑满血封装。 * 包含：空转逻辑、物理缓动、异步结算、多ID自适应。 */ (function() { // 1. 系统核心配置 const SYS_CONFIG = { BASE_BET: 5.5, // 准入门槛 WIN_RATE: 2.0, // 赔率 ANIM_DURATION: 5000, // 物理动画时长 (5s) MIN_SPIN: 12, // 最小旋转圈数 EASING: 'cubic-bezier(0.15, 0, 0.1, 1)' // 工业级物理缓动曲线 }; // 2. 状态机 (State Engine) let terminalState = { wallet: 146.8, // 账户余额 isLocked: false, // 运行锁 targetSide: null, // 预设侧 (龙/虎) rotationStack: 0, // 角度累加 sessionMode: 'IDLE' // 状态：IDLE(闲置), REAL(实战), VOID(空转) }; // 3. 启动初始化 window.onload = () => { console.log("%c ⚡ 计划书级别代码已加载 | 逻辑摘除完成 ", "background:#000;color:#f00;padding:5px;"); refreshUI(); autoBindControls(); }; // 4. 自动化控制绑定 (适配所有老代码 ID) function autoBindControls() { // 扫描执行按钮 const fire = document.getElementById('exec-btn') || document.getElementById('main-btn') || document.querySelector('.play-btn'); if (fire) fire.addEventListener('click', handleFireAction); // 扫描交互侧按钮 ['DRAGON', 'TIGER'].forEach(side => { const el = document.getElementById('opt-' + side) || document.getElementById('b-' + side); if (el) el.onclick = () => toggleSelection(side); }); } // 5. 交互逻辑：选择、切换与取消 function toggleSelection(side) { if (terminalState.isLocked) return; // 核心：点击已选则变为“空转模式” if (terminalState.targetSide === side) { terminalState.targetSide = null; } else { terminalState.targetSide = side; } updateButtonVisuals(); // 触感同步 if (window.navigator.vibrate) window.navigator.vibrate(30); } // 6. 核心引擎：执行抽取 (含空转逻辑判定) async function handleFireAction() { if (terminalState.isLocked) return; // 判定当前模式 const isRealPlay = (terminalState.targetSide !== null); // 准入逻辑 if (isRealPlay) { if (terminalState.wallet < SYS_CONFIG.BASE_BET) { pushAlert("⚠️ 系统提示：可用余额不足以支持当前量化。"); return; } // 实战扣费逻辑 terminalState.wallet -= SYS_CONFIG.BASE_BET; terminalState.sessionMode = 'REAL'; } else { terminalState.sessionMode = 'VOID'; console.log("🌀 执行空转流程..."); } // 视觉响应：先转起来，不等待后端 terminalState.isLocked = true; refreshUI(); runWheelAnimation(); // 结果生成 (异步保底) const roundResult = await fetchOutcome(); // 结算同步 setTimeout(() => { processFinalResult(roundResult, isRealPlay); }, SYS_CONFIG.ANIM_DURATION); } // 7. 物理动画引擎 function runWheelAnimation() { const wheel = document.getElementById('wheel') || document.querySelector('.wheel-img'); if (!wheel) return; const randomOffset = Math.floor(Math.random() * 360); terminalState.rotationStack += (360 * SYS_CONFIG.MIN_SPIN) + randomOffset; wheel.style.transition = `transform ${SYS_CONFIG.ANIM_DURATION}ms ${SYS_CONFIG.EASING}`; wheel.style.transform = `rotate(${terminalState.rotationStack}deg)`; } // 8. 结算中心 function processFinalResult(winSide, wasReal) { terminalState.isLocked = false; const sideCN = (winSide === 'DRAGON' ? '龙' : '虎'); if (wasReal) { // 实战模式结算 if (terminalState.targetSide === winSide) { const bonus = SYS_CONFIG.BASE_BET * SYS_CONFIG.WIN_RATE; terminalState.wallet += bonus; pushAlert(`🎉 预测命中！\n开出结果：[${sideCN}]\n资金已返还。`, "success"); } else { pushAlert(`❌ 预测未命中。\n开出结果：[${sideCN}]\n量化波动损耗。`, "fail"); } } else { // 空转模式结算 pushAlert(`🎡 观摩局结束。\n盘面指向：[${sideCN}]\n(空转未参与博弈)`, "info"); } refreshUI(); } // 9. 辅助工具逻辑 function fetchOutcome() { return new Promise(res => { setTimeout(() => res(Math.random() < 0.5 ? 'DRAGON' : 'TIGER'), 600); }); } function refreshUI() { const b = document.getElementById('balance'); if (b) b.innerText = terminalState.wallet.toFixed(1); } function updateButtonVisuals() { document.querySelectorAll('.opt, .bet-btn').forEach(btn => { btn.classList.remove('selected', 'active'); if (terminalState.targetSide && btn.id.includes(terminalState.targetSide)) { btn.classList.add('selected', 'active'); } }); } function pushAlert(m, t) { if (window.customMsg) window.customMsg(m, t); else alert(m); } })();
