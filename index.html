(function() { // =============================== // 顶级配置：对齐 99.9% 的细节 // =============================== const CORE = { U_UNIT: 100, ENTRY_FEE: 550, // 5.5 WHEEL_REVS: 22, T_CONST: 5600, SECTOR_ANGLE: 45, BOUNCE_DEG: 1.5, SPIN_FALLBACK: 300 // transition 兜底 }; let session = { active: false, side: null, totalDeg: 0, settled: false, isPending: localStorage.getItem('_q_pending') === 'true' }; // =============================== // 启动 // =============================== window.onload = () => { console.log( "%c 🛰️ 艺术品级内核加载 | 实战硬核对齐 ", "color:#ed1c24;font-weight:bold;" ); initStorage(); if (session.isPending) recoverSession(); initCore(); bindVisibilityGuard(); }; // =============================== // 选择逻辑 // =============================== function handleSelect(s) { if (session.active) return; const prev = session.side; session.side = (session.side === s) ? null : s; if (prev !== session.side) { vibrate(25); syncStyles(); } } // =============================== // 执行引擎 // =============================== async function startTrade() { if (session.active) return; const isReal = session.side !== null; let bal = getBal(); if (isReal && bal < CORE.ENTRY_FEE) { toast("⚠️ 指令拒绝：资金池核验未通过"); return; } session.active = true; session.settled = false; localStorage.setItem('_q_pending', 'true'); try { const data = await secureHandshake(isReal); if (isReal) { bal -= CORE.ENTRY_FEE; setBal(bal); } applyPhysics(data.win, isReal); } catch (e) { console.error(e); unlock(); toast("📡 通讯链路异常"); } } // =============================== // 物理惯性 + 回弹 // =============================== function applyPhysics(target, real) { const wheel = document.querySelector('.wheel-img') || document.getElementById('wheel'); if (!wheel) { unlock(); return; } const base = (target === 'DRAGON') ? 0 : CORE.SECTOR_ANGLE; const offset = (Math.random() * 30) - 15; session.totalDeg += (360 * CORE.WHEEL_REVS) + base + offset; wheel.style.transition = `transform ${CORE.T_CONST}ms cubic-bezier(0.2,0,0,1)`; wheel.style.transform = `rotate(${session.totalDeg}deg)`; let ended = false; const finish = () => { if (ended) return; ended = true; // 回弹 const bounce = (Math.random() > 0.5 ? 1 : -1) * CORE.BOUNCE_DEG; wheel.style.transition = `transform 280ms ease-out`; wheel.style.transform = `rotate(${session.totalDeg + bounce}deg)`; setTimeout(() => finalize(target, real), 320); }; wheel.addEventListener('transitionend', finish, { once: true }); setTimeout(finish, CORE.T_CONST + CORE.SPIN_FALLBACK); } // =============================== // 结算（幂等） // =============================== function finalize(win, real) { if (session.settled) return; session.settled = true; unlock(); const isWin = (session.side === win); if (real && isWin) { setBal(getBal() + CORE.ENTRY_FEE * 2); } const label = win === 'DRAGON' ? '🐉 龙' : '🐯 虎'; if (real) { isWin ? alert(`🎊 量化成功: [${label}]`) : alert(`📉 量化波动: [${label}]`); } else { alert(`🎡 观摩结束: [${label}]`); } refreshDisplay(); } // =============================== // 恢复未完成订单（刷新跑单） // =============================== function recoverSession() { console.warn("⚠️ 检测到未完成订单，已安全回收"); localStorage.setItem('_q_pending', 'false'); session.active = false; session.settled = false; } // =============================== // 样式同步 // =============================== function syncStyles() { document.querySelectorAll('[data-side], .opt').forEach(el => { const side = el.dataset.side || el.id?.split('-')[1]; el.classList.toggle('active', session.side === side); el.classList.toggle('selected', session.side === side); }); } // =============================== // 通讯模拟 // =============================== function secureHandshake(real) { return new Promise(res => { setTimeout(() => { res({ win: Math.random() < 0.5 ? 'DRAGON' : 'TIGER' }); }, 700); }); } // =============================== // UI / 工具 // =============================== function toast(msg) { console.log(msg); alert(msg); } function vibrate(ms) { if (navigator.vibrate) navigator.vibrate(ms); } function unlock() { session.active = false; localStorage.setItem('_q_pending', 'false'); } function refreshDisplay() { const el = document.getElementById('balance'); if (el) el.innerText = (getBal() / CORE.U_UNIT).toFixed(1); } function getBal() { return parseInt(localStorage.getItem('_q_bal'), 10); } function setBal(v) { localStorage.setItem('_q_bal', String(v)); } function initStorage() { if (!localStorage.getItem('_q_bal')) { localStorage.setItem('_q_bal', '14680'); } } function initCore() { const fire = document.getElementById('exec-btn') || document.getElementById('main-btn'); if (fire) fire.onclick = startTrade; ['DRAGON','TIGER'].forEach(s => { const el = document.querySelector(`[data-side="${s}"]`) || document.getElementById('opt-' + s); if (el) el.onclick = () => handleSelect(s); }); refreshDisplay(); } function bindVisibilityGuard() { document.addEventListener('visibilitychange', () => { if (document.hidden && session.active) { console.warn("⚠️ 页面切后台，强制解锁"); unlock(); } }); } })();
