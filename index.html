function onExecute() { const cost = state.bets[state.bIdx]; if(state.rolling || !state.pick || state.bal < cost) return; state.rolling = true; state.bal -= cost; document.getElementById('val-bet').innerText = cost.toFixed(2); document.getElementById('val-win').innerText = "0.00"; document.getElementById('wheel').classList.add('blur'); update(); /* ===== 大转盘：只负责转，不参与龙虎判定 ===== */ const winIdx = Math.floor(Math.random()*8); const rounds = state.speed===3 ? 4 : 8; const target = (360 - (state.deg%360)) + rounds*360 - (winIdx*45); state.deg += target; const wheel = document.getElementById('wheel'); wheel.style.transitionDuration = state.speed===3 ? '0.8s' : state.speed===2 ? '2s' : '4s'; wheel.style.transform = `rotate(${state.deg}deg)`; /* ===== 龙虎结果：独立生成 ===== */ const dragonTigerResult = Math.random() > 0.5 ? 'DRAGON' : 'TIGER'; setTimeout(() => { wheel.classList.remove('blur'); state.rolling = false; if(state.pick === dragonTigerResult){ const winAmt = cost * 2; state.bal += winAmt; document.getElementById('val-win').innerText = winAmt.toFixed(2); const cell = document.getElementById('cell-'+dragonTigerResult); cell.classList.add('winner'); setTimeout(()=>cell.classList.remove('winner'),1500); } update(); if(state.auto) setTimeout(onExecute, 1200); }, parseFloat(wheel.style.transitionDuration)*1000); }
