const idle = new Audio('audio/idle.mp3'); const intro = new Audio('audio/spin_intro.mp3'); const loop = new Audio('audio/spin_loop.mp3'); const slow = new Audio('audio/spin_slow.mp3'); const win = new Audio('audio/win_hit.mp3'); const lose = new Audio('audio/lose_cut.mp3'); idle.loop = true; loop.loop = true; // 初始化：进入等待状态 function enterIdle() { stopAll(); idle.volume = 0.4; idle.play(); } // 点击开始 function startSpin() { stopAll(); intro.play(); setTimeout(() => { loop.play(); }, 1200); } // 减速 function slowDown() { loop.pause(); slow.play(); } // 出结果 function showResult(multiplier) { slow.pause(); if (multiplier === 0) { lose.play(); } else { win.play(); } // 结果结束后回到 idle setTimeout(() => { enterIdle(); }, 2000); } // 工具函数 function stopAll() { [idle, intro, loop, slow, win, lose].forEach(a => { a.pause(); a.currentTime = 0; }); }
