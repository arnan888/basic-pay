(function() { // --- 1. 基础账目 (分制，杜绝小数浮点误差) --- let store = { get: () => parseInt(localStorage.getItem('_bal') || "14680"), set: (v) => localStorage.setItem('_bal', v.toString()) }; let state = { active: false, side: null, deg: 0 }; window.onload = () => { refreshUI(); // 绑定选择逻辑 ['DRAGON', 'TIGER'].forEach(s => { const el = document.getElementById('opt-' + s) || document.querySelector(`[data-side="${s}"]`); if (el) el.onclick = () => { if (state.active) return; state.side = (state.side === s) ? null : s; // 同步视觉：选中变红，取消变灰 document.querySelectorAll('.opt, [data-side]').forEach(b => { const bSide = b.id?.split('-')[1] || b.dataset.side; b.style.filter = (state.side === bSide) ? 'brightness(1.2) drop-shadow(0 0 5px red)' : 'none'; }); }; }); // 绑定执行逻辑 const btn = document.getElementById('exec-btn') || document.getElementById('main-btn'); if (btn) btn.onclick = start; }; async function start() { if (state.active) return; const isReal = state.side !== null; let currentBal = store.get(); // 【第一道关卡：硬核验证】 if (isReal) { if (currentBal < 550) { alert("❌ 余额不足 (需5.5)"); return; } // 验证通过，立即扣费，绝不拖泥带水 currentBal -= 550; store.set(currentBal); refreshUI(); } // 【第二道关卡：锁定引擎】 state.active = true; // 预定结果 (0:龙, 1:虎) const winIdx = Math.random() < 0.5 ? 0 : 1; const winSide = winIdx === 0 ? 'DRAGON' : 'TIGER'; // 【第三道关卡：强制动画对齐】 await new Promise(resolve => { const wheel = document.getElementById('wheel') || document.querySelector('.wheel-img'); const targetDeg = winIdx * 45; // 假设0度是龙，45度是虎 state.deg += (360 * 15) + targetDeg + (Math.random() * 4 - 2); wheel.style.transition = 'transform 5s cubic-bezier(0.15, 0, 0, 1)'; wheel.style.transform = `rotate(${state.deg}deg)`; setTimeout(resolve, 5200); // 略长于动画时间，确保停稳 }); // 【第四道关卡：结算逻辑】 if (isReal) { if (state.side === winSide) { store.set(store.get() + 1100); // 赢了加 11.0 alert(`🎊 命中：${winSide === 'DRAGON' ? '龙' : '虎'}`); } else { alert(`💔 未命中：${winSide === 'DRAGON' ? '龙' : '虎'}`); } } else { alert(`🎡 空转结果：${winSide === 'DRAGON' ? '龙' : '虎'}`); } state.active = false; refreshUI(); } function refreshUI() { const el = document.getElementById('balance'); if (el) el.innerText = (store.get() / 100).toFixed(1); } })();
